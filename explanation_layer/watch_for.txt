from .utils import (
    get_active_assets,
    get_top_positive_risk_contributors,
    find_capped_assets,
)


def generate_watch_for(
    weights,
    risk_contributions,
    max_weight_constraint=0.35,
):
    bullets = []

    active_assets = get_active_assets(weights)

    top_weights = sorted(
        active_assets.items(),
        key=lambda x: x[1],
        reverse=True,
    )

    top_risk = get_top_positive_risk_contributors(
        risk_contributions,
        top_n=3,
    )

    capped = set(
        find_capped_assets(
            weights,
            max_weight_constraint=max_weight_constraint,
        )
    )

    used = set()

    # 1 — capped first
    for asset in top_weights:
        ticker = asset[0]

        if ticker in capped:
            bullets.append(
                f"Watch {ticker}: it already reached the weight limit, so the portfolio depends heavily on it."
            )
            used.add(ticker)
            break

    # 2 — biggest weight
    for ticker, w in top_weights:
        if ticker in used:
            continue

        bullets.append(
            f"Watch {ticker}: it is one of the largest positions, so big moves here may affect the whole portfolio."
        )
        used.add(ticker)
        break

    # 3 — biggest risk (if different)
    for ticker, rc in top_risk:
        if ticker in used:
            continue

        bullets.append(
            f"Watch {ticker}: it is a major source of portfolio risk."
        )
        break

    return bullets[:2]
